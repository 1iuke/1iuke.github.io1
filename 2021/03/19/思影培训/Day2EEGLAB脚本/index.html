EEG脚本编程。数据批处理

0 数据

采集每个被试的脑电数据；使用64个电极，每个电极记录n =3000 毫秒数据；分为36 个epoch

个体数据（64，3000，36）

EEG.data (channel,time,epoch)

1 批处理

同时处理多个被试数据。

假设我们有100个数据 被试的文件名为1.set 2.set .... 100.set

%% 1导入原始数据 
%注意  是导入bp数据的函数 
%如果你的数据是别的厂家别的格式 要自己导入一次 查看EEG.history中的代码
%对于每个被试
for i = 1:100
    %拼接被试的文件名 
    rawname = [num2str(i),'.vhdr'];
    setname = [num2str(i),'.set'];
    %使用函数导入数据 路径记得修改 不要忘记后面的分隔符\
    EEG = pop_loadbv('路径1\', rawname, [], []);
    %检查EEG结构体是否完好
    EEG = eeg_checkset( EEG );
    %把set文件存到磁盘上
    EEG = pop_saveset( EEG, 'filename',setname,'filepath','路径2\');
end

%% 2通道定位
for i = 1:100
    %拼接被试的文件名 
    setname = [num2str(i),'.set'];
    %导入set格式文件
    EEG = pop_loadset('filename',setname,'filepath','路径2\');
    %通道定位
    EEG=pop_chanedit(EEG, 'lookup','坐标文件自己生成路径');
    %去除无用电极
    EEG = pop_select( EEG,'nochannel',{'HEO' 'VEOG'});
    %进行0.1 - 40Hz 的带通滤波
    EEG = pop_eegfiltnew(EEG, 0.1, 40, 33000, 0, [], 0);
    %进行48-52 Hz的凹陷滤波
    EEG = pop_eegfiltnew(EEG, 48, 52, 1650, 1, [], 0);
    %降采样到500Hz
    EEG = pop_resample( EEG, 500);
    %分段
    EEG = pop_epoch( EEG, {  '10'  '11'  }, [-1  2], 'newname', ' resampled epochs', 'epochinfo', 'yes');
    %基线校正
    EEG = pop_rmbase( EEG, [-1000     0]);
    %提取正确反应试次
    EEG = pop_selectevent( EEG, 'type',200,'deleteevents','off','deleteepochs','on','invertepochs','off');
    %把set文件存到磁盘上
    EEG = pop_saveset( EEG, 'filename',setname,'filepath','路径3\');
end

%% 手动插值坏导 替换坏段
%要一个一个被试查看 不能批处理 
%做好的数据存在路径4中
EEG = pop_interp(EEG, [13  60], 'spherical');
EEG = pop_rejepoch( EEG, [2 77 78] ,0);

%% RunICA
for i = 1:100
    %拼接被试的文件名 
    setname = [num2str(i),'.set'];
    %导入set格式文件
    EEG = pop_loadset('filename',setname,'filepath','路径4\');
    %如果数据中没有对坏电极进行插值，则运行此行
%     EEG = pop_runica(EEG, 'extended',1,'interupt','on');
    %如果你进行了坏电极插值，则需要执行这行
    %此处的m为当前电极的数量 n为插值的电极数量
    EEG = pop_runica(EEG, 'extended',1,'pca',m-n,'interupt','on');
    %把set文件存到磁盘上
    EEG = pop_saveset( EEG, 'filename',setname,'filepath','路径5\');
end
%% 手动查看独立成分 并去除眨眼等伪迹成分
%不能批处理 数据存在路径6
%% 极端值去伪迹、重参考、基线校正
for i = 1:100
    %拼接被试的文件名 
    setname = [num2str(i),'.set'];
    %导入set格式文件
    EEG = pop_loadset('filename',setname,'filepath','路径6\');
    % 此处是+-100 
    EEG = pop_eegthresh(EEG,1,[1:62] ,-100,100,-1,1.998,0,1);
    %重参考
    EEG = pop_reref( EEG, [33 43] );
    %基线校正
    EEG = pop_rmbase( EEG, [-1000     0]);
    %把set文件存到磁盘上
    EEG = pop_saveset( EEG, 'filename',setname,'filepath','路径7\');
end

%% 手动浏览数据 再次筛查伪迹
%结果存在路径8

3 matlab code1

3.1 计算个体水平

将每个被试的epoch进行平均，此时EEG.data由（59，3000，36）变为（59，3000）

将平均后的数据组合起来，放入EEG.avg中，保存每个被试的个体平均。数据形状为（10，59，3000）

clear all; clc; close all

%% part1: compute group-level ERP

Subj = [1:10]; %% subject numbers
for i = 1:length(Subj)
    setname = strcat(num2str(i),'_LH.set'); %% filename of set file
    setpath = 'D:\MyWorkSpace\Matlab\SiYingPeiXun\25EEG_day2\Example_data\';
    EEG = pop_loadset('filename',setname,'filepath',setpath); %% load the data
    EEG = eeg_checkset( EEG );
    % 汇总每个被试个体水平叠加平均好的数据
    % 10 59 3000  sub  chl time
    EEG_avg(i,:,:) = squeeze(mean(EEG.data,3)); %% single-subject ERPs; EEG_avg dimension: subj*channel*time
end

save('Group_level_ERP.mat','EEG_avg');  %% save the data of subjects

3.2 计算组水平

计算是所有被试Cz电极所有时间点的组水平平局数据

选取所有被试的Cz电极数据（10，1，3000）进行平均；

数据形状为（1，1，3000）

squeeze后变为（3000，1）

%% part2: plot group-level ERP

Cz = 13; % select the channel to plot (display maximum response)
figure;
% 提取所有被试Cz电极所有时间点的数据
% mean_data 是Cz电极所有时间点的组水平平局数据
mean_data = squeeze(mean(EEG_avg(:,Cz,:),1)); %% select the data at Cz, average across subjects, mean_data: 1*3000

plot(EEG.times, mean_data,'k','linewidth', 1.5); %% plot the waveforms
set(gca,'YDir','reverse'); %% reverse the direction of Y axis
axis([-500 1000 -15 10]);  %% define the region to display
% xlim([-500 1000]);  %% define the region of X axis
% ylim([-15 10]); %% define the region of Y axis
title('Group-level at Cz','fontsize',16); %% specify the figure name
xlabel('Latency (ms)','fontsize',16); %% name of X axis
ylabel('Amplitude (uV)','fontsize',16);  %% name of Y axis

3.3 绘制特定点地形图

选取波峰波谷附近数据 绘制地形图

%% part3: plot the scalp maps at dominant peaks

N2_peak = 207; P2_peak = 374; %% dominant peaks on waveforms
N2_interval = find((EEG.times>=197)&(EEG.times<=217)); %% define the N2 intervals [peak-10 peak+10]
P2_interval = find((EEG.times>=364)&(EEG.times<=384)); %% define the P2 intervals [peak-10 peak+10]

N2_amplitude = squeeze(mean(EEG_avg(:,:,N2_interval),3));  %% N2 amplitude for each subject and each channel
P2_amplitude = squeeze(mean(EEG_avg(:,:,P2_interval),3));   %% P2 amplitude for each subject and each channel

figure; 
subplot(121); topoplot(mean(N2_amplitude),EEG.chanlocs,'maplimits',[-15 15]); title('N2 Amplitude','fontsize',16); %% N2 scalp map (group-level)
subplot(122); topoplot(mean(P2_amplitude),EEG.chanlocs,'maplimits',[-15 15]); title('P2 Amplitude','fontsize',16); %% P2 scalp map (group-level)

3.4 绘制感兴趣时间地形图

%% part4: series of scalp mps

time_interval = [0:100:500]; %% specify the time intervals to display (to be changed)
figure; 
for i = 1:length(time_interval)
    latency_range = [time_interval(i) time_interval(i)+100]; %% lower and upper limits
    latency_idx = find((EEG.times>=latency_range(1))&(EEG.times<=latency_range(2))); %% interval of the specific regions
    Amplitude = squeeze(mean(mean(EEG_avg(:,:,latency_idx),1),3)); %% 1*channel (averaged across subjects and interval)
    subplot(2,3,i); 
    topoplot(Amplitude,EEG.chanlocs,'maplimits',[-10 10]); %% topoplot(Amplitude,EEG.chanlocs);
    setname = strcat(num2str(latency_range(1)),'--',num2str(latency_range(2)),'ms'); %% specify the name of subplots
    title(setname,'fontsize',16); %% display the names of subplots
end

5 matlab code3

差异性判断

T检验

5.1 比较单个个体L3 L4 时间上的差异

clear all; clc; close all

Subj = [1:10]; 
Cond = {'L1','L2','L3','L4'};

%% compute averaged data
for i = 1:length(Subj)
    setname = strcat(num2str(i),'_LH.set'); 
    setpath = 'D:\MyWorkSpace\Matlab\SiYingPeiXun\25EEG_day2\Example_data\';
    EEG = pop_loadset('filename',setname,'filepath',setpath); 
    EEG = eeg_checkset( EEG );
    for j = 1:length(Cond)
        EEG_new = pop_epoch( EEG, Cond(j), [-1  2], 'newname', 'Merged datasets pruned with ICA', 'epochinfo', 'yes'); 
        EEG_new = eeg_checkset( EEG_new );
        EEG_new = pop_rmbase( EEG_new, [-1000     0]); 
        EEG_new = eeg_checkset( EEG_new );
        EEG_avg(i,j,:,:)=squeeze(mean(EEG_new.data,3));  %% subj*cond*channel*timepoints
    end 
end

%%  point-by-point paried t-test  across multiple time points 
data_test = squeeze(EEG_avg(:,:,13,:)); %% select the data at Cz, data_test: subj*cond*time
for i = 1:size(data_test,3)
    data_1 = squeeze(data_test(:,3,i)); %% select condition L3 for each time point
    data_2 = squeeze(data_test(:,4,i)); %% select condition L4 for each time point
    % 检验两个条件的差异  用配对样本T检验  ttest
    % 检验两组样本之间的差异，要做独立样本T检验  ttest2
    [h p] = ttest(data_1,data_2); %% ttest comparison
    P_ttest(i) = p; %% save the p value from ttest
end
figure; 
subplot(211); plot(EEG.times,squeeze(mean(data_test(:,3,:),1)),'b'); %% plot the average waveform for Condition L3
hold on; plot(EEG.times,squeeze(mean(data_test(:,4,:),1)),'r'); %% plot the average waveform for Condition L4
subplot(212); plot(EEG.times,P_ttest); ylim([0 0.05]); %%plot the p values from ttest

5.2 比较不同通道的差异

%% point-by-point paried t-test  across multiple channels
% 定义感兴趣的成分 --> 锁定成分
test_idx = find((EEG.times>=197)&(EEG.times<=217)); %% define the intervals
data_test = squeeze(mean(EEG_avg(:,:,:,test_idx),4)); %% select the data in [197 217]ms, subj*cond*channel
for i = 1:size(data_test,3)
    data_1 = squeeze(data_test(:,3,i)); %% select condition L3 for each channel
    data_2 = squeeze(data_test(:,4,i)); %% select condition L4 for each channel
    [h,p,ci,stats] = ttest(data_1,data_2); %% ttest comparison
    P_ttest2(i) = p; %% save the p value from ttest
    T_ttest2(i) = stats.tstat; 
end

figure; 
subplot(141); 
topoplot(squeeze(mean(data_test(:,3,:),1)),EEG.chanlocs,'maplimits',[-20 20]); 
subplot(142); 
topoplot(squeeze(mean(data_test(:,4,:),1)),EEG.chanlocs,'maplimits',[-20 20]); 
subplot(143); 
topoplot(T_ttest2,EEG.chanlocs); 
subplot(144); 
topoplot(P_ttest2,EEG.chanlocs,'maplimits',[0 0.05]);

%% point-by-point repeated measures of ANOVA across time points
data_test = squeeze(EEG_avg(:,:,13,:)); %% select the data at Cz, data_test: subj*cond*time
for i = 1:size(data_test,3)
    data_anova = squeeze(data_test(:,:,i)); %% select the data at time point i
    [p, table] = anova_rm(data_anova,'off');  %% perform repeated measures ANOVA
    P_anova(i) = p(1); %% save the data from ANOVA
end

mean_data = squeeze(mean(data_test,1)); %% dimension: cond*time
figure; 
subplot(211);plot(EEG.times, mean_data,'linewidth', 1.5); %% waveform for different condition 
set(gca,'YDir','reverse');
axis([-500 1000 -35 25]);
subplot(212);plot(EEG.times,P_anova); axis([-500 1000 0 0.05]); %% plot the p values from ANOVA
